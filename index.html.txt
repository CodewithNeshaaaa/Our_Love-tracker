  <script>
    function updateTime() {
      const proposalDate = new Date("2023-02-09T20:00:00");
      const now = new Date();

      let years = now.getFullYear() - proposalDate.getFullYear();
      let months = now.getMonth() - proposalDate.getMonth();
      let days = now.getDate() - proposalDate.getDate();
      let hours = now.getHours() - proposalDate.getHours();

      if (hours < 0) {
        hours += 24;
        days--;
      }
      if (days < 0) {
        const prevMonth = new Date(now.getFullYear(), now.getMonth(), 0).getDate();
        days += prevMonth;
        months--;
      }
      if (months < 0) {
        months += 12;
        years--;
      }

      document.getElementById("timePassed").textContent =
        `${years} years, ${months} months, ${days} days, and ${hours} hours`;
    }

    updateTime();
    setInterval(updateTime, 3600000); // update every hour
  </script>
